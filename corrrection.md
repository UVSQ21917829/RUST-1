```markdown
fn pgcd(mut x :i32,mut y:i32)->i32{
         let mut i :i32=0;
         while i==0{
                    if x>y{
                          x=x-y;  
                          if x==0{
                          i=y;
                          }
                          }else {
                                y=y-x; 
                                if y==0{
                                       i=x;
                                       }
                                }
                          }
           return i;
                                    }
```
